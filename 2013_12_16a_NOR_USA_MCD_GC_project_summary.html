The meeting that I prepared this outline for did not go as I hoped. Director Murray questioned the validity of non-GC data as training, and suggested that David Philips had already been down the road of USA MCD. I finally caught up with DP and got the story of what had been done before and why it was abandoned, and decided that this direction is still valuable. Valubale for two reasons. There will be plenty of settings where we have MCD but not linked hospital data; and there is a lot more MCD data than we can ever dream of having gold-standard hospital data.

So I have returned to this and come up with some results on the Septicimia analysis that look much more promising than the ones DP had before he put things on the back burner.

This is prototyped in notebook 2014_01_09_UCS_MCD_GC_septicimia, and now I am going to refactor it into a python script that I can run on the cluster.

%%file knn_mcd_correct.py

import pandas as pd, glob, numpy as np
dname = '/clustertmp/abie/USA_MCD/'

from sklearn.feature_extraction.text import HashingVectorizer
vectorizer = HashingVectorizer(ngram_range=(1,2))

import sklearn.neighbors
clf = sklearn.neighbors.KNeighborsClassifier(n_neighbors=10, weights='uniform')

# TODO: go straight from DataFrame to sparse array (should be faster)
def extract_training_str(s):
    t = ' '.join([s['cause_entity%d'%(i+1)] for i in range(20)])
    return t

import sys
print sys.argv
age, sex, year = sys.argv[1:]

# tell pandas not to do anything too smart with the cause lists
converters = {}
for i in range(20):
    converters['cause_entity%d'%(i+1)] = str
    converters['cause_record%d'%(i+1)] = str

converters['underlying_cause'] = str

df = pd.DataFrame()
for fname in glob.glob(dname + 'all-%s-%s-%s.csv'%(age,sex,year)):
    df = df.append(pd.read_csv(fname, converters=converters, low_memory=False), ignore_index=True, )
    
import json
map_icd = json.load(file('map_icd.json'))

df['cod'] = df.underlying_cause.map(map_icd)


train_rows = {}
test_rows = {}

for j, df_j in df.groupby('cod'):
    #print j, len(df_j.index)
    
    if j != '_gc':
        train_rows[j] = list(df_j.index)
    else:
        # icd9 septicemia codes
        codes = '038 0380 0381 03810 03811 03812 03819 0382 0383 0384 03840 03841 03842 03843 03844 03849 0388 0389'.split()
        # icd10 septicemia codes
        codes += 'A41 A410 A4101 A4102 A411 A412 A413 A414 A415 A4150 A4151 A4152 A4153 A4159 A418 A4181 A4189 A419'.split()
        rows = df_j.underlying_cause.isin(codes)
        test_rows[j] = list(df_j[rows].index)
        
        
# train


y_train = []
X_train_str = []

for j, rows_j in train_rows.iteritems():
    if len(rows_j) == 0:
        continue
        
    y_train += [j]*len(rows_j)
    X_train_str += list(df.ix[rows_j].apply(extract_training_str, axis=1))
    
X_train = vectorizer.transform(X_train_str)

clf.fit(X_train, y_train)


# test

X_test = {}

for j, rows_j in test_rows.iteritems():
    if len(rows_j) == 0:
        continue
    X_test_str = df.ix[rows_j].apply(extract_training_str, axis=1)
    
    X_test[j] = vectorizer.transform(X_test_str)
    
    #print j, len(rows_j), X_test[j].shape


y_pred = {}

for j in X_test.keys():
    y_pred[j] = {}
    # TODO: make this faster, by predicting chunks together
    # (predicting all together was giving memory error)
    for r in range(X_test[j].shape[0]):
        y_pred[j][r] = clf.predict_proba(X_test[j][r])
    #print j, y_pred[j].shape
    
j = '_gc'
for i in range(4):
    s = pd.Series(y_pred[j][i][0], index=clf.classes_)
    print s[s>0].order(ascending=False)
    print
    
    

j = '_gc'
s = pd.Series(np.zeros_like(clf.classes_, dtype=float), index=clf.classes_)
for i in range(X_test[j].shape[0]):
    s += pd.Series(y_pred[j][i][0], index=clf.classes_)

fname = 'septicemia_redist_%s_%s_%s.csv' %(age,sex,year)
fname = fname.replace('*','-')  # files with * in the filename are bad
s.to_csv(dname + fname)
